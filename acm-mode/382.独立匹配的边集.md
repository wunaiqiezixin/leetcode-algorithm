# 382.独立匹配的边集

### ***C++***
```CPP
#include <iostream>
#include <vector>
using namespace std;
//边结构体：方便阅读
struct Edge
{
    int node1;
    int node2;
};
 
int main()
{
    //输入输出优化
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    //读入边
    int n, m;
    cin >> n >> m;
    vector<Edge> Edges(m + 1);
    int i;
    for (i = 1; i <= m; ++i) cin >> Edges[i].node1;
    for (i = 1; i <= m; ++i) cin >> Edges[i].node2;
 
    int q;//测试组数
    cin >> q;
    while (q--)
    {
        bool isValid = true;//该边集是否合法
        //将边集中的边全部标记
        vector<bool> is_in_set(m + 1, false);
        //节点是否在边集中
        vector<bool> is_node_in_set(n + 1, false);
         
        //读入边集的所有边，如果有节点重复出现，该边集不合法
        int e;
        cin >> e;
        while (e--)
        {
            int order;
            cin >> order;
            //标记边
            is_in_set[order] = true;
            //判断该边的节点是否已经出现在边集中
            if (is_node_in_set[Edges[order].node1] || 
                is_node_in_set[Edges[order].node2])
                isValid = false;
            //将节点标记，代表出现在边集
            is_node_in_set[Edges[order].node1] = true;
            is_node_in_set[Edges[order].node2] = true;
        }
        //如果边集中的边都合法
        if (isValid)
        {
            //判断不属于边集的边
            for (i = 1; i <= m; ++i)
            {
                if (!is_in_set[i])
                {
                    int node1 = Edges[i].node1;
                    int node2 = Edges[i].node2;
                    if (is_node_in_set[node1] &&
                        is_node_in_set[node2])
                    {
                        isValid = false;
                        break;
                    }
                }
            }
        }
        if (isValid) cout << "Yes" << "\n";
        else cout << "No" << "\n";
    }
     
    return 0;
}
```
