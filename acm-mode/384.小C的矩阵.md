# 384.小C的矩阵

### ***C++***
```cpp
/*
脑筋急转弯 + 排序不等式
*/
#include <iostream>
#include <vector>
#include <cstdint>
#include <algorithm>
using namespace std;

int n, m;// n x m 的矩阵

// 第i行的权重 x 第i行元素的和
// 第j列的权重 x 第j列元素的和

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    //权重
    vector<int> rowWeight(n);
    vector<int> colWeight(m);
    //每一行/列的和
    vector<uint64_t> rowSum(n, 0);
    vector<uint64_t> colSum(m, 0);

    //统计每一行、每一列的和
    int i, j;
    for (i = 0; i < n; ++i) cin >> rowWeight[i];
    for (j = 0; j < m; ++j) cin >> colWeight[j];
    for (i = 0; i < n; ++i)
        for (j = 0; j < m; ++j) 
        {
            int value;
            cin >> value;
            rowSum[i] += value;
            colSum[j] += value;
        }
    
    uint64_t result = 0;
    //排序
    sort(rowWeight.begin(), rowWeight.end());
    sort(rowSum.begin(), rowSum.end());
    sort(colWeight.begin(), colWeight.end());
    sort(colSum.begin(), colSum.end());

    for (i = 0; i < n; ++i)
        result += static_cast<uint64_t>(rowWeight[i]) * rowSum[i];
    for (j = 0; j < m; ++j)
        result += static_cast<uint64_t>(colWeight[j]) * colSum[j];
    
    cout << result << "\n";

    return 0;
}
```
