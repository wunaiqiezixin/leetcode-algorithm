# 131. 分割回文串
[力扣题目链接](https://leetcode.cn/problems/palindrome-partitioning/description/)

---
## 🧠思路
分割问题实际上是组合问题   
将变量```startIndex```定为切割线   
问题就变成了在字符串```s```中放切割线的问题   
用一个for循环横向遍历
```C
for (int i = startIndex; i < strlen(s); i++)
```
```startIndex```为一条已经确定的切割线，```i```为这一次要确定的切割线   
于是切出来的子串为```s[startIndex, i]```    
如果不为回文，就直接```continue```,跳过这一分支   
如果为回文，就进行回溯搜索：   
用一个二维数组```char** path```记录回文子串，用一个三维数组```char*** ans```记录组合     
将切割出来的回文子串加入```path```     
然后从```i+1```进行```backtracking()```    
最后将回文子串从```path```弹出进行回溯   

如果```startIndex >= strlen(s)```，说明已经找到了一个符合条件的组合，收集结果   
```C
    //终止条件：
```

### ***C***
```C
/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
bool isPalindrome(char* s, int start, int end)
{
    for (int i = start, j = end; i < j; i++, j--)
    {
        if (s[i] != s[j]) return false;
    }
    return true;
}

//二维数组path记录回文子串
//三维数组ans记录组合
void backtracking(char* s, int startIndex, char*** ans, int* ansTop, char** path, int pathTop, int* colSizes)
{
    //终止条件：
    if (startIndex >= strlen(s))
    {
        //拷贝path
        char** tmp = (char**)malloc(sizeof(char*) * pathTop);
        for (int i = 0; i < pathTop; i++) tmp[i] = path[i];
        colSizes[*ansTop] = pathTop;
        ans[(*ansTop)++] = tmp;
        return;
    }
    //回溯搜索多叉树
    for (int i = startIndex; i < strlen(s); i++)
    {
        //子串为s[startIndex, i]
        if (isPalindrome(s, startIndex, i))
        {
            //拷贝子串
            char* tmp = (char*)malloc(sizeof(char) * (i - startIndex + 2));
            for (int j = 0; j < (i - startIndex + 1); j++) tmp[j] = s[startIndex + j];
            tmp[i - startIndex + 1] = '\0';
            path[pathTop++] = tmp;//将回文子串加入路径
            backtracking(s, i + 1, ans, ansTop, path, pathTop, colSizes);
            pathTop--;//回溯
        }
        else continue;//不是回文，跳过树的分支
    }
}
char*** partition(char* s, int* returnSize, int** returnColumnSizes) {
    //分配空间
    char*** ans = (char***)malloc(sizeof(char**) * 100001);
    *returnSize = 0;
    *returnColumnSizes = (int*)malloc(sizeof(int) * 100001);
    char** path = (char**)malloc(sizeof(char*) * 10001);
    backtracking(s, 0, ans, returnSize, path, 0, *returnColumnSizes);
    return ans;
}
```

```CPP
class Solution {
private:
    vector<vector<string>> ans;
    vector<string> path;
    void backtracking(const string& s, int startIndex)
    {
        if (startIndex >= s.size())
        {
            ans.push_back(path);
            return;
        }

        for (int i = startIndex; i < s.size(); i++)
        {
            if (isValid(s, startIndex, i))
            {
                string tmp = s.substr(startIndex, i - startIndex + 1);
                path.push_back(tmp);
                backtracking(s, i + 1);
                path.pop_back();
            }
            //continue;
        }
    }
    bool isValid(const string& s, int start, int end)
    {
        for (int i = start, j = end; i < j; i++, j--)
        {
            if (s[i] != s[j])
                return false;
        }
        return true;
    }
public:
    vector<vector<string>> partition(string s) {
        backtracking(s, 0);
        return ans;
    }
};
```

>动态规划求回文子串

```CPP
class Solution {
private:
    vector<vector<bool>> isPalindrome;
    vector<vector<string>> ans;
    vector<string> path;
    void backtracking(const string& str, int startIndex)
    {
        if (startIndex >= str.size())
        {
            ans.push_back(path);
            return;
        }
        for (int i = startIndex; i < str.size(); ++i)
        {
            string Sub = str.substr(startIndex, i - startIndex + 1);
            if (!isPalindrome[startIndex][i])
                continue;
            path.push_back(Sub);
            backtracking(str, i + 1);
            path.pop_back();
        }
    }
    void computePalindrome(const string& s)
    {
        isPalindrome.resize(s.size(), vector<bool>(s.size(), false));
        int i, j;
        for (i = s.size() - 1; i >= 0; --i)
        {
            for (j = i; j < s.size(); ++j)
            {
                if (j == i) isPalindrome[i][j] = true;
                else if (j - i == 1) isPalindrome[i][j] = (s[i] == s[j]);
                else {isPalindrome[i][j] = (s[i] == s[j] && isPalindrome[i + 1][j - 1]);}
            }
        }
    }
public:
    vector<vector<string>> partition(string s) {
        ans.clear();
        path.clear();
        computePalindrome(s);
        backtracking(s, 0);
        return ans;
    }
};
```
