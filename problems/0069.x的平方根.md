# 69.x的平方根
[力扣题目链接](https://leetcode.cn/problems/sqrtx/description/)

## 牛顿迭代法

### ***C***
```C
#define EPS 1e-7
int mySqrt(int x) {
    if (x == 0) return 0;
    double a = x / 2.0;
    double last;
    do {
        last = a;
        a = (a + x / a) / 2;
    } while (fabs(last - a) > EPS);
    return (int)a;
}
```

## 二分查找
```CPP
class Solution {
private:
    const int SQRT_INT_MAX = sqrt(INT_MAX);
public:
    int mySqrt(int x) {
        // 满足 m * m <= x 的m的最大值
        int left = 0, right = min(x, SQRT_INT_MAX);//左闭右闭
        while (left <= right)
        {
            int m = left + (right - left) / 2;
            if (m * m <= x)    //如果 m * m = x, 继续查找右边，期盼更大的满足条件的 m
                left = m + 1;
            else
                right = m - 1;
        }
        return right;
    }
};
```
