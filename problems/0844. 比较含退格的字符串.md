# 844. 比较含退格的字符串
[题目链接](https://leetcode.cn/problems/backspace-string-compare/description/)

---
## 思路
>双指针

### ***C++***
```CPP
class Solution {
public:
    bool backspaceCompare(string s, string t) {
        //双指针
        int ps = s.size() - 1, pt = t.size() - 1;
        //记录'#'出现的次数
        int sSkip = 0, tSkip = 0;
        for (;;)
        {
            //处理字符串s的退格
            while (ps >= 0)
            {
                if (s[ps] == '#')
                    sSkip++;
                else if (sSkip)
                    sSkip--;
                else 
                    break;
                ps--;
            }
            //处理字符串t的退格
            while (pt >= 0)
            {
                if (t[pt] == '#')
                    tSkip++;
                else if (tSkip)
                    tSkip--;
                else 
                    break;
                pt--;
            }
            //比较
            if (ps < 0 && pt < 0)
                return true;
            if (ps < 0 || pt < 0)
                return false;
            if (s[ps] != t[pt])
                return false;
            ps--;
            pt--;
        }
        return true;
    }
};
```

>栈

### ***C++***
```CPP
class Solution {
private:
    string backspace(const string& str)
    {
        string ret;
        for (auto& ch : str)
        {
            if (ch == '#' && ret.size())
                ret.pop_back();//退格符，栈顶元素出栈
            else if (ch == '#' && !ret.size())
                continue;//栈为空，无法退格，跳过
            else 
                ret += ch;//入栈
        }
        return ret;
    }
public:
    bool backspaceCompare(string s, string t) {
        return backspace(s) == backspace(t);
    }
};
```
