# 76. 最小覆盖子串
[力扣题目链接](https://leetcode.cn/problems/minimum-window-substring/description/)

---
## 思路
>滑动窗口

这是一道典型的滑动窗口问题
定义两个变量`ans_left = -1, ans_right = s.size()`记录满足条件的最短子串的下标
定义两个变量`left, right`表示当前滑动窗口的左右边界

1. 怎么样叫做当前的滑动窗口满足条件(状态)
>s的子串s[left, right]中所有字符的出现次数大于字符串t中字符的出现次数
```CPP
        //默认初始化为0
        int cnt_t[128]{};
        int cnt_s[128]{};
```
```CPP
    //是否涵盖字符串t(当前的滑动窗口是否满足条件？)
    bool isCovered(int cnt_s[], int cnt_t[])//128对应一个字节(char)
    {
        int i;
        for (i = 'A'; i <= 'Z'; ++i)
            if (cnt_s[i] < cnt_t[i])
                return false;
        for (i = 'a'; i <= 'z'; ++i)
            if (cnt_s[i] < cnt_t[i])
                return false;
        return true;
    }
```


2. 如果当前的滑动窗口**满足状态**：(更新结果,并移动左边界试图寻找更短的子串)
```CPP

```


### ***C++***
```CPP
class Solution {
private:
    //是否涵盖字符串t(当前的滑动窗口是否满足条件？)
    bool isCovered(int cnt_s[], int cnt_t[])//128对应一个字节(char)
    {
        int i;
        for (i = 'A'; i <= 'Z'; ++i)
            if (cnt_s[i] < cnt_t[i])
                return false;
        for (i = 'a'; i <= 'z'; ++i)
            if (cnt_s[i] < cnt_t[i])
                return false;
        return true;
    }
public:
    string minWindow(string s, string t) {
        int ans_left = -1, ans_right = s.size();//最短子串的左右下标
        int left, right;//滑动窗口的左右边界
        left = 0;
        //默认初始化为0
        int cnt_t[128]{};
        int cnt_s[128]{};
        //统计字符串t中的字符个数
        for (const char ch : t)
            cnt_t[ch]++;
        //移动滑动窗口右边界
        for (right = 0; right < s.size(); ++right)
        {
            //将字符加入滑动窗口
            cnt_s[s[right]]++;
            //如果涵盖(滑动窗口满足条件)
            while (isCovered(cnt_s, cnt_t))
            {
                if (ans_right - ans_left > right - left)//找到了更短的子串
                {
                    //记录
                    ans_left = left;
                    ans_right = right;
                }
                //再移动右边界已经没有意义:减枝
                //移动左边界，试图寻找更短的符合条件的子串
                cnt_s[s[left]]--;
                left++;
            }
            //如果不涵盖，继续移动右边界，直到涵盖为止
        }
        //判断是否找到了子串并返回
        return ans_left < 0 ? "" : s.substr(ans_left, ans_right - ans_left + 1);
    }
};
```

```CPP
class Solution {
public:
    string minWindow(string s, string t) {
        /* ASII
           A~Z: 65~90
           a~z: 97~122
        */
        // 0~128 覆盖了所有英文字母的ASII码, 空间复杂度 O(1)
        int cnt_s[128]{};
        int cnt_t[128]{};

        int need = 0;   // 字符串 t 中的字母种类数量
        int valid = 0;  // 字符串 s 中覆盖 t 中的字母的个数

        for (char ch : t) {
            if (cnt_t[ch]++ == 0) {
                need++;
            }
        }

        int ans_left = -1, ans_right = s.size();

        int left = 0;
        int right;
        // 滑动窗口循环，时间复杂度 O(n + m)
        for (right = 0; right < s.size(); right++) {
            char ch1 = s[right];
            cnt_s[ch1]++;
            if (cnt_s[ch1] == cnt_t[ch1]) {
                valid++;
            }
            while (valid == need) {
                if (ans_right - ans_left > right - left) {
                    ans_left = left;
                    ans_right = right;
                }
                char ch2 = s[left];
                cnt_s[ch2]--;
                if (cnt_s[ch2] < cnt_t[ch2]) {
                    valid--;
                }
                left++;
            }
        }
        return ans_left < 0 ? "" : s.substr(ans_left, ans_right - ans_left + 1);
    }
};
```
