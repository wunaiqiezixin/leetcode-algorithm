# 904. 水果成篮
[力扣题目链接](https://leetcode.cn/problems/fruit-into-baskets/description/)

---
>滑动窗口


### ***C++***
```CPP
class Solution {
public:
    int totalFruit(vector<int>& fruits) {
        //怎么样叫做滑动窗口满足状态?
        unordered_map<int, int> cnt;// 统计fruits中的每种水果种类采摘的次数
        int left, right;
        left = 0;
        int result = 0;//最大数目
        //移动窗口的右边界
        for (right = 0; right < fruits.size(); ++right)
        {
            //将该水果种类加入滑动窗口
            cnt[fruits[right]]++;
            //如果滑动窗口不满足状态
            while (cnt.size() > 2)//采摘的种类大于2
            {
                //再移动右边界已经毫无意义
                //移动窗口的左边界
                cnt[fruits[left]]--;
                if (cnt[fruits[left]] == 0) 
                    cnt.erase(fruits[left]);//如果为0，将元素移除
                left++;
            }
            result = max(result, right - left + 1);
        }
        return result;
    }
};
```
